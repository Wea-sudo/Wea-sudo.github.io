Easy 难度:
1.AtCoder-abc336_a Long Loong
做题思路：
先打印L，在输出n个o，最后在输出ng
AC代码：
#include
#include
using namespace std;

int main()
{
int n;
cin >> n;
cout << ‘L’;
for (int i = 1; i <= n; i++)
{
cout << ‘o’;
}
cout << “ng”;

}
2.CodeForces-1703A YES or YES
做题思路：
先把字符串全部变为小写，再与”yes”进行比较
AC代码：
#include
#include
#include
using namespace std;

int main()
{
int n;
scanf(“%d”, &n);
for (int i = 0; i < n; i++)
{
char a[4];
scanf(“ %s”,&a);
for (int i = 0; i < 4; i++)
{
a[i] = tolower(a[i]);

    }
    
    if (!strcmp("yes", a)) cout << "YES" << endl;
    else cout << "NO" << endl;
}

}
3.洛谷-P2955 Even? Odd? G
做题思路：先用字符串存储数字，判断最后一位数的奇偶性
AC代码：
#include
#include
using namespace std;

int main()
{
int n;
cin >> n;
for (int i = 0; i < n; i++)
{
char a[60];
cin >> a;
int len = strlen(a);
if ((a[len-1]-‘0’) % 2 == 0) cout << “even” << endl;
else cout << “odd” << endl;
}

}

1.CodeForces-1980A
做题思路：
先统计每个难度的题数（不超过m个），在统计总共要出的题数，相减得出结果
AC代码：
#include
#include
using namespace std;

int main()
{
int t;
cin >> t;
for (int i = 1; i <= t; i++)
{
int n, m;
cin >> n >> m;//n数量，m举办的轮数
map<char, int>a;
for (char j = ‘A’; j <= ‘G’; j++)
{
a[j] = 0;
}
string s;
cin >> s;
for (int j = 0; j < n; j++)
{
if (a[s[j]] < m) a[s[j]]++;

    }
    int cnt = m*7;
    for (char j = 'A'; j <= 'G'; j++)
    {
        cnt -= a[j];
    }
    cout << cnt << endl;
}

}
2.洛谷-B4012
做题思路：
先计算每天遵守该规矩的人数，判断该天该规矩是否符合民意，在计算该规矩符合民意的天数，得出该规矩是正确还是错误的
AC代码：
#include
#include
#include
using namespace std;

int main()
{
int n,m,cnt=0;//n人数，m天数
long long k;
cin >> n >> m >> k;
for (int i = 1; i <= m; i++)
{
int mount=0;
for (int j = 1; j <= n; j++)
{
long long ki;
cin >> ki;
if (ki == k)mount++;
}
if (2 * mount >= n) cnt++;
}
if (2 * cnt >= m)cout << “YES”;
else cout << “NO”;
}

1.AtCoder-abc342_c Many Replacement
做题思路：
先把每一个字母最后会变成什么记录下来，最后循环把字符串每个字母对应的最后的字母打印出来
AC代码：
#include
#include
#include
#include
using namespace std;

int main()
{
int n;
cin >> n;
string s;
cin >> s;
int q;
cin >> q;
vectorc;
vectord;

for (int i =0; i <q; i++)
{
    char c1, d1;
    cin >> c1 >> d1;
    c.push_back(c1);
    d.push_back(d1);
}
map<char, char>alpha;
for (char i = 'a'; i <= 'z'; i++)
{
    
    alpha[i] = i;
    for (int j = 0; j < q; j++)

}

}

2.洛谷-B3977 更好的交换
解题思路：
不直接交换，记下每行交换的顺序（矩阵内每个数字对应的行列仍不变），打印出交换后对应行列的数字
AC代码：
#include
#include
#include
#include
using namespace std;

int main()
{
int n, m;
cin >> n >> m;
vector<vector> a(n,vector(n));

for (int i = 0; i < n; i++)
{
    for (int j = 0; j < n; j++)
    {
        cin >> a[i][j];
    }
}
vector<int> x(n),y(n);
for (int i = 0; i < n; i++)
{
    x[i] = i;
    y[i] = i;
}
for (int i = 0; i < m; i++)
{
    int choice, z1, z2;
    cin >> choice >> z1 >> z2;
    z1--;
    z2--;
    if (choice)
    {
        int t = x[z1];
        x[z1] = x[z2];
        x[z2] = t;
    }
    else
    {
        int t = y[z1];
        y[z1] = y[z2];
        y[z2] = t;
    }
}
for (int i = 0; i <n; i++)
{
    for (int j = 0; j <n; j++)
    {
        cout << a[x[i]][y[j]]<<' ';
    }
    printf("\n");
}

}

本专题总结：
对于某些题目不一定要把过程按照题目要求的操作，可以考虑最后的状态
要善于运用stl库，对于某些功能可以先查找stl库是否有（如交换，排序等）
